pipeline {
    agent any
    parameters {
        choice(name: 'BRANCH_NAME', choices: ['master', 'Branch1'], description: 'Enter branch name to execute ?')
    }
    stages {
        stage('parallel') {
            parallel {
                stage('STAGE-1') {
                    when {
                        expression{ params.BRANCH_NAME == "master" }
                    }
                    steps{
                        sh "echo STAGE-2 executes if branch is master"
                    }
                }

                stage('STAGE-2'){
                    when {
                        expression{ params.BRANCH_NAME == "master" }
                    }
                    steps{
                        sh "echo STAGE-2 executes if branch is master"
                    }
                }
            }    
        }    
    }
}
